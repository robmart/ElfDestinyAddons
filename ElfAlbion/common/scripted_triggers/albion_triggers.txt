has_expedition_amount_trigger = {
    custom_description = {
        text = has_expedition_amount_trigger
        subject = root
        object = $LOCATION$
        value = $AMOUNT$
        $LOCATION$ = {
            trigger_if = {
                limit = {
                    exists = var:expeditions_completed_here
                }
                var:expeditions_completed_here >= $AMOUNT$
            }
            trigger_else = {
                always = no
            }
        }
    }
}
