give_human_lover_secret_or_nothing_effect = {
	$CHARACTER$ = {
		if = {
			limit = {
				NOR = {
					has_trait = human_lover
					any_secret = { secret_type = secret_human_lover }
				}
				is_adult = yes
				relation_with_human_is_bad_for_me_or_my_liege = yes
			}
			add_secret = {
				type = secret_human_lover
			}
		}
	}
}
