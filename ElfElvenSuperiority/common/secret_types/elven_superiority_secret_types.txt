secret_human_lover = {
	category = deviancy

	is_valid = {
		scope:secret_owner = {
			NOT = {
				has_trait = human_lover
			}
		}
	}

	is_criminal = { 
		scope:secret_owner = {
			relation_with_human_is_bad_for_me_or_my_liege = yes
		}
	}

	on_expose = {
		save_scope_as = secret

		# secret_exposed_notification_effect = yes
		
		scope:secret_owner = {
			if = {
				limit = {
					NOT = {
						this = scope:secret_exposer
					}
				}
				save_scope_as = owner
				trigger_event = elf.secrets.0001
			}
		}
		if = {
			limit = {
				scope:secret_owner = {
					relation_with_human_is_bad_for_me_or_my_liege = yes
				}
			}

			scope:secret_owner = {
				add_trait = human_lover
			}
		}

	}
}
